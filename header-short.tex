\newcommand\denselyConnectNodes[2]{
  \foreach\n [count=\lyrIdx, remember=\lyrIdx as \previdx] in #2 {
    \foreach\y in {1,...,\n} {
      \ifnum\lyrIdx>1
        \foreach\x in {1,...,\prevn}
          \draw[->] (#1-\previdx-\x) -- (#1-\lyrIdx-\y);
      \fi
    }
    \xdef\prevn{\n}
  }
}
