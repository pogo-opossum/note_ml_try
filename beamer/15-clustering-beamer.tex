\section{Introduction to Clustering}

\begin{frame}{What Is Clustering?}
  \framesubtitle{Unsupervised learning without labels}
  \begin{itemize}
    \item \alert{Clustering} is a central task in unsupervised learning.
    \item Goal: organize data into meaningful groups \emph{without} relying on ground-truth labels.
    \item Sharp contrast to classification: no known targets, no external supervision to validate
          proposed partitions.
    \item The learning is fundamentally \alert{exploratory}: uncover latent structure, patterns,
          or groupings implicit in the geometry of observations.
  \end{itemize}
  \begin{block}{Practical applications}
    Clustering serves as a preliminary step for data understanding, feature engineering,
    or as a component in larger pipelines---for instance, to compress data, initialize mixture
    models, or detect anomalies.
  \end{block}
\end{frame}

\begin{frame}{Clustering vs.\ Classification}
  \begin{columns}[t]
    \column{0.48\textwidth}
      \begin{block}{Classification (supervised)}
        \begin{itemize}
          \item Training guided by \alert{known targets}.
          \item Objective: maximize predictive accuracy.
          \item External validation available.
          \item Well-defined success metric.
        \end{itemize}
      \end{block}
    \column{0.48\textwidth}
      \begin{block}{Clustering (unsupervised)}
        \begin{itemize}
          \item Data arrive \alert{unlabeled}.
          \item Objective: discover hidden structure.
          \item No external supervision.
          \item Success depends on domain goals.
        \end{itemize}
      \end{block}
  \end{columns}
  \vspace{0.6em}
  \begin{alertblock}{Key insight}
    In clustering, there is no universal ``correct'' answer. The quality of a clustering
    depends on the choice of distance function, the inductive bias imposed, and the
    specific application context.
  \end{alertblock}
\end{frame}

\section{Problem Setup and Formalization}

\begin{frame}{Clustering Problem Setup}
  \framesubtitle{Formalizing the task}
  \begin{itemize}
    \item Let $\mathbf{X}=(\mathbf{x}_1,\ldots,\mathbf{x}_n)$ be a dataset of $n$ points in
          $\mathbb{R}^d$.
    \item A clustering task seeks to partition the dataset into $k$ clusters
          $C_1,\ldots,C_k$.
    \item Points in the same cluster should be \alert{similar}; points in different clusters
          should be \alert{dissimilar}.
    \item Precise meanings of ``similar'' and ``dissimilar'' are not universal: they depend on
          distance function and inductive bias.
  \end{itemize}
  \begin{block}{Optimization perspective}
    Rather than purely combinatorial partitioning, clustering is often viewed as an
    \alert{optimization problem}: define a quantitative notion of cluster quality and search
    for the grouping that optimizes it.
  \end{block}
\end{frame}

\begin{frame}{Inductive Biases in Clustering}
  \begin{itemize}
    \item Different clustering methods embody different inductive biases about what constitutes
          a ``good'' cluster:
      \begin{itemize}
        \item \alert{Compactness}: clusters as tight spherical regions (e.g., $k$-means).
        \item \alert{Separation}: clusters well-separated in feature space.
        \item \alert{Density}: clusters as regions of high point density.
        \item \alert{Connectivity}: clusters as connected regions.
      \end{itemize}
    \item For the methods discussed in this presentation, similarity is expressed primarily
          in terms of \alert{distances in $\mathbb{R}^d$}.
  \end{itemize}
\end{frame}

\begin{frame}{Cluster Representation: Prototypes}
  \framesubtitle{Where are clusters located?}
  \begin{itemize}
    \item Each cluster is summarized by a \alert{prototype} $\mathbf{m}_j\in\mathbb{R}^d$,
          for $j=1,\ldots,k$.
    \item Intuitively, $\mathbf{m}_j$ plays the role of a \alert{representative point} for
          cluster $C_j$.
    \item In many algorithms, it corresponds to the ``center'' of the cluster.
    \item The precise meaning of ``center'' depends on the loss function and the metric in use.
  \end{itemize}
  \begin{block}{Geometric interpretation}
    Prototypes make the location of clusters explicit in feature space, allowing us to
    visualize and interpret the clustering geometrically.
  \end{block}
\end{frame}

\begin{frame}{Cluster Representation: Assignment Matrix}
  \framesubtitle{Which points belong to which cluster?}
  \begin{itemize}
    \item To encode membership, each data point $\mathbf{x}_i$ is associated with an
          \alert{assignment vector} $\mathbf{r}_i\in\{0,1\}^k$.
    \item The entry $r_{ij}$ indicates whether $\mathbf{x}_i$ belongs to cluster $C_j$:
  \end{itemize}
  \[
  r_{ij}=
  \begin{cases}
  1 & \text{if } \mathbf{x}_i \text{ is assigned to cluster } C_j\\
  0 & \text{otherwise.}
  \end{cases}
  \]
  \begin{alertblock}{Hard assignment (1-of-$k$ encoding)}
    For each $i$, exactly one component of $\mathbf{r}_i$ equals $1$ and all others equal $0$.
    Every point is assigned to one and only one cluster. Later, probabilistic models will
    relax this to \alert{soft assignments}.
  \end{alertblock}
\end{frame}

\section{Clustering Paradigms}

\begin{frame}{Two Broad Paradigms}
  \framesubtitle{Partitional vs.\ hierarchical clustering}
  \begin{itemize}
    \item Although many clustering families exist, most classical methods fall into two
          broad paradigms:
      \begin{enumerate}
        \item \alert{Partitional clustering}
        \item \alert{Hierarchical clustering}
      \end{enumerate}
    \item The difference is not only algorithmic; it also affects:
      \begin{itemize}
        \item The kind of output the method produces.
        \item What questions it is naturally suited to answer.
      \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}{Partitional Clustering}
  \begin{itemize}
    \item Aims directly at producing a \alert{single, flat partition} of the dataset into $k$
          non-overlapping clusters.
    \item The number $k$ is typically fixed in advance (or selected by model selection criteria).
    \item The algorithm searches for a clustering that optimizes a cost function $J$ measuring
          cluster quality.
    \item Clustering closely resembles \alert{parameter estimation}: simultaneously choosing
          prototypes and assignments.
  \end{itemize}
  \begin{block}{Key advantages}
    \begin{itemize}
      \item \alert{Computational scalability}: effective on large datasets with appropriate
            implementations.
      \item Typical representatives: $k$-means, $k$-medoids, mixture models.
    \end{itemize}
  \end{block}
\end{frame}

\begin{frame}{Hierarchical Clustering}
  \begin{itemize}
    \item Does not return a single flat partition.
    \item Produces a \alert{nested family of partitions} organized as a tree.
    \item Commonly visualized as a \alert{dendrogram}.
    \item Moving along the tree changes the granularity of the description:
      \begin{itemize}
        \item Bottom: many small clusters.
        \item Top: clusters merge (agglomerative) or split (divisive).
      \end{itemize}
  \end{itemize}
  \begin{block}{Key advantages}
    \begin{itemize}
      \item No need to commit to a specific $k$ at the outset.
      \item Any desired number of clusters obtained by ``cutting'' the dendrogram at a chosen height.
      \item Richer description, often more informative.
    \end{itemize}
  \end{block}
\end{frame}

\begin{frame}{Hierarchical Clustering: Limitations}
  \begin{alertblock}{Computational cost}
    Typically higher computational cost than partitional methods. Classic agglomerative
    algorithms have $O(n^3)$ time complexity in the worst case.
  \end{alertblock}
  \vspace{0.4em}
  \begin{alertblock}{Irrevocable decisions}
    Hierarchical decisions made early in the process \alert{cannot be revised later}.
    Makes the method sensitive to:
    \begin{itemize}
      \item Choice of linkage criterion.
      \item Choice of distance metric.
    \end{itemize}
  \end{alertblock}
\end{frame}

\begin{frame}{Choosing Between Paradigms}
  \begin{itemize}
    \item The two paradigms are \alert{not in competition}: they are complementary.
    \item Appropriate choice depends on the goals of the analysis.
  \end{itemize}
  \begin{block}{When to prefer partitional methods}
    \begin{itemize}
      \item Speed and scalability matter.
      \item Reasonable prior belief about the number of clusters.
      \item Large datasets where hierarchical methods are prohibitive.
    \end{itemize}
  \end{block}
  \begin{block}{When to prefer hierarchical methods}
    \begin{itemize}
      \item Primary goal is to understand multi-scale structure.
      \item Number of clusters $k$ is genuinely unknown.
      \item Desire a richer picture of data relationships.
    \end{itemize}
  \end{block}
\end{frame}

\section{The $k$-Means Algorithm}

\begin{frame}{Introduction to $k$-Means}
  \framesubtitle{The workhorse of partitional clustering}
  \begin{itemize}
    \item Among partitional clustering methods, \alert{$k$-means} occupies a special place.
    \item Reasons for its prominence:
      \begin{itemize}
        \item Simplicity.
        \item Speed.
        \item Clarity of interpretation.
      \end{itemize}
    \item Despite being conceptually straightforward, it embodies many themes that recur in
          more advanced unsupervised learning algorithms.
  \end{itemize}
  \begin{block}{Key themes}
    Optimization under discrete assignments, alternating minimization, convergence to local
    optima, and sensitivity to initialization. Understanding $k$-means provides a strong
    foundation for more sophisticated methods.
  \end{block}
\end{frame}

\begin{frame}{The $k$-Means Objective}
  \framesubtitle{Within-cluster sum of squares}
  \begin{itemize}
    \item Given dataset $\mathbf{X}$ and target number of clusters $k$, $k$-means searches
          simultaneously for:
      \begin{itemize}
        \item Assignment matrix $R=(r_{ij})$
        \item Set of prototypes $M=(\mathbf{m}_1,\ldots,\mathbf{m}_k)$
      \end{itemize}
    \item that minimize the following objective:
  \end{itemize}
  \[
  J(R,M)=\sum_{i=1}^{n}\sum_{j=1}^{k} r_{ij}\|\mathbf{x}_i-\mathbf{m}_j\|^2.
  \]
  \begin{block}{Interpretation}
    This is the \alert{within-cluster sum of squares}. It measures how tightly each cluster
    concentrates around its prototype. Each point contributes only through its squared
    distance to the prototype of its own cluster.
  \end{block}
\end{frame}

\begin{frame}{Inductive Bias of $k$-Means}
  \begin{itemize}
    \item The objective makes the inductive bias of $k$-means explicit.
    \item Minimizing $J$ encourages \alert{compact} clusters in Euclidean space.
    \item Points should lie close to their cluster center.
    \item A good clustering is one in which most squared distances $\|\mathbf{x}_i-
          \mathbf{m}_j\|^2$ (for the assigned $j$) are small.
  \end{itemize}
  \begin{alertblock}{When $k$-means works well}
    This bias fits well when clusters are roughly \alert{spherical}---or, more precisely,
    when cluster structure is well captured by Euclidean distance and means.
  \end{alertblock}
  \begin{alertblock}{Limitations}
    Becomes restrictive in the presence of elongated, curved, or density-based cluster shapes.
  \end{alertblock}
\end{frame}

\begin{frame}{Lloyd's Algorithm: Structure}
  \framesubtitle{Alternating optimization}
  \begin{itemize}
    \item The standard algorithm for $k$-means is \alert{Lloyd's algorithm}.
    \item It is an instance of \alert{coordinate descent} or \alert{alternating minimization}.
    \item Basic idea: optimize over one set of variables while holding the other fixed,
          then switch roles.
  \end{itemize}
  \begin{block}{Two-phase iteration}
    \begin{enumerate}
      \item \alert{Assignment step}: Fix prototypes $M$, optimize assignments $R$.
      \item \alert{Update step}: Fix assignments $R$, optimize prototypes $M$.
    \end{enumerate}
    Repeat until convergence.
  \end{block}
\end{frame}

\begin{frame}{Lloyd's Algorithm: Assignment Step}
  \begin{itemize}
    \item Given current prototypes $\mathbf{m}_1,\ldots,\mathbf{m}_k$, the assignment
          step minimizes $J$ with respect to $R$.
    \item For each point $\mathbf{x}_i$, we need to choose which cluster to assign it to.
    \item Since $r_{ij}\in\{0,1\}$ and $\sum_j r_{ij}=1$, we have:
  \end{itemize}
  \[
  J = \sum_{i=1}^{n} \sum_{j=1}^{k} r_{ij}\|\mathbf{x}_i-\mathbf{m}_j\|^2
    = \sum_{i=1}^{n} \|\mathbf{x}_i-\mathbf{m}_{c(i)}\|^2,
  \]
  where $c(i)$ is the cluster to which $\mathbf{x}_i$ is assigned.
  \begin{block}{Optimal assignment}
    Assign each point to its \alert{nearest prototype}:
    \[
    c(i) = \arg\min_{j=1,\ldots,k} \|\mathbf{x}_i-\mathbf{m}_j\|^2.
    \]
  \end{block}
\end{frame}

\begin{frame}{Lloyd's Algorithm: Update Step}
  \begin{itemize}
    \item Given current assignments, the update step minimizes $J$ with respect to $M$.
    \item For a fixed cluster $j$, let $C_j$ be the set of points assigned to it.
    \item The contribution to $J$ from cluster $j$ is:
  \end{itemize}
  \[
  \sum_{i: c(i)=j} \|\mathbf{x}_i-\mathbf{m}_j\|^2.
  \]
  \begin{itemize}
    \item This is minimized when $\mathbf{m}_j$ is the \alert{mean} of the points in $C_j$:
  \end{itemize}
  \[
  \mathbf{m}_j = \frac{1}{|C_j|}\sum_{i\in C_j}\mathbf{x}_i.
  \]
  \begin{block}{Optimal prototypes}
    Update each prototype to be the \alert{centroid} (arithmetic mean) of its assigned points.
  \end{block}
\end{frame}

\begin{frame}{Lloyd's Algorithm: Pseudocode}
  \begin{enumerate}
    \item \textbf{Initialize} prototypes $\mathbf{m}_1,\ldots,\mathbf{m}_k$ (e.g., randomly
          from data points).
    \item \textbf{Repeat} until convergence:
      \begin{enumerate}
        \item \textbf{Assignment step}: For each $i=1,\ldots,n$, set
              \[
              c(i) \leftarrow \arg\min_{j=1,\ldots,k} \|\mathbf{x}_i-\mathbf{m}_j\|^2.
              \]
        \item \textbf{Update step}: For each $j=1,\ldots,k$, set
              \[
              \mathbf{m}_j \leftarrow \frac{1}{|C_j|}\sum_{i\in C_j}\mathbf{x}_i.
              \]
      \end{enumerate}
    \item \textbf{Return} final prototypes and assignments.
  \end{enumerate}
  \vspace{0.4em}
  \begin{alertblock}{Convergence criterion}
    Stop when assignments no longer change, or when improvement in $J$ is below a threshold.
  \end{alertblock}
\end{frame}

\begin{frame}{Convergence of Lloyd's Algorithm}
  \framesubtitle{Monotonic decrease of the objective}
  \begin{itemize}
    \item Each iteration of Lloyd's algorithm is guaranteed to \alert{not increase} $J$.
    \item The assignment step minimizes $J$ over $R$ for fixed $M$.
    \item The update step minimizes $J$ over $M$ for fixed $R$.
    \item Therefore: $J^{(t+1)} \leq J^{(t)}$ at every iteration.
  \end{itemize}
  \begin{block}{Finite convergence}
    Since there are finitely many possible assignments and $J$ decreases monotonically,
    the algorithm must \alert{converge in a finite number of steps}.
  \end{block}
  \begin{alertblock}{Convergence to local optimum}
    Convergence is guaranteed, but only to a \alert{local minimum}. Different initializations
    can lead to different solutions.
  \end{alertblock}
\end{frame}

\begin{frame}{Cost Curve Over Iterations}
  \begin{center}
    \includegraphics[width=0.6\textwidth]{figures/kmeansobj.pdf}
  \end{center}
  \begin{itemize}
    \item Cost function $J(R,M)$ decreases monotonically and stabilizes as the algorithm
          converges.
    \item Steep descent in first few iterations reflects large reorganizations.
    \item Subsequent plateau indicates marginal adjustments near a fixed point.
  \end{itemize}
\end{frame}

\begin{frame}{Computational Complexity}
  \begin{itemize}
    \item Let $n$ = number of points, $d$ = dimensionality, $k$ = number of clusters,
          $t$ = number of iterations.
    \item \alert{Assignment step}: For each of $n$ points, compute distance to each of $k$
          prototypes.
      \begin{itemize}
        \item Cost: $O(nkd)$ per iteration.
      \end{itemize}
    \item \alert{Update step}: Compute mean of points in each cluster.
      \begin{itemize}
        \item Cost: $O(nd)$ per iteration.
      \end{itemize}
    \item \alert{Overall}: $O(nkdt)$ for $t$ iterations.
  \end{itemize}
  \begin{block}{Practical efficiency}
    In practice, $t$ is typically small (often $< 100$), and the algorithm scales well
    to large datasets, especially with optimized implementations.
  \end{block}
\end{frame}

\begin{frame}{Initialization Strategies}
  \framesubtitle{Mitigating sensitivity to starting points}
  \begin{itemize}
    \item The quality of the final clustering depends heavily on initialization.
    \item Poor initialization can lead to suboptimal local minima.
  \end{itemize}
  \begin{block}{Common strategies}
    \begin{enumerate}
      \item \alert{Random selection}: Choose $k$ data points uniformly at random as initial
            prototypes. Simple but can be unlucky.
      \item \alert{Multiple random starts}: Run Lloyd's algorithm multiple times with
            different random initializations and select the solution with lowest $J$.
      \item \alert{$k$-means++}: A smarter initialization that spreads out initial prototypes.
            Provides provable approximation guarantees.
    \end{enumerate}
  \end{block}
\end{frame}

\begin{frame}{$k$-Means++ Initialization}
  \begin{itemize}
    \item Proposed by Arthur \& Vassilvitskii (2007).
    \item Key idea: Choose initial prototypes that are \alert{far apart} from each other.
  \end{itemize}
  \begin{block}{Algorithm}
    \begin{enumerate}
      \item Choose the first prototype uniformly at random from data points.
      \item For each subsequent prototype:
        \begin{itemize}
          \item For each data point $\mathbf{x}_i$, compute distance $D(\mathbf{x}_i)$ to
                nearest already-chosen prototype.
          \item Choose next prototype with probability proportional to $D(\mathbf{x}_i)^2$.
        \end{itemize}
      \item Proceed with standard Lloyd's algorithm.
    \end{enumerate}
  \end{block}
  \begin{alertblock}{Guarantee}
    $k$-means++ is $O(\log k)$-competitive with the optimal clustering in expectation.
  \end{alertblock}
\end{frame}

\begin{frame}{Empty Clusters}
  \framesubtitle{A practical issue}
  \begin{itemize}
    \item During iterations, it is possible for a cluster to become \alert{empty}: no points
          are assigned to it.
    \item If this happens, the corresponding prototype update is undefined (division by zero).
  \end{itemize}
  \begin{block}{Common solutions}
    \begin{itemize}
      \item \alert{Remove the cluster}: Reduce $k$ by one and continue.
      \item \alert{Reassign the prototype}: Choose a new prototype (e.g., the point farthest
            from its current cluster center).
      \item \alert{Split a large cluster}: Find the cluster with highest internal variance
            and split it.
    \end{itemize}
  \end{block}
  \begin{itemize}
    \item Good initialization (e.g., $k$-means++) makes empty clusters less likely.
  \end{itemize}
\end{frame}

\section{Model Selection for $k$-Means}

\begin{frame}{Choosing the Number of Clusters $k$}
  \framesubtitle{A fundamental challenge}
  \begin{itemize}
    \item In many applications, the number of clusters $k$ is \alert{not known in advance}.
    \item We need a principled way to select $k$ from data.
    \item Unlike supervised learning, there is no validation accuracy to optimize.
    \item Various heuristics and criteria have been proposed.
  \end{itemize}
  \begin{alertblock}{No universal solution}
    Model selection for clustering remains more art than science. Different criteria can
    suggest different values of $k$, and domain knowledge often plays a crucial role.
  \end{alertblock}
\end{frame}

\begin{frame}{The Elbow Method}
  \framesubtitle{Visual heuristic for choosing $k$}
  \begin{itemize}
    \item Run $k$-means for a range of values $k=1,2,\ldots,k_{\max}$.
    \item Plot the within-cluster cost $J(k)$ as a function of $k$.
    \item Because adding more clusters increases flexibility, cost necessarily decreases as
          $k$ grows.
    \item However, there typically exists a point beyond which improvement becomes marginal.
  \end{itemize}
  \begin{block}{The ``elbow''}
    The point where the curve starts to flatten is taken as a reasonable compromise between
    fit quality and model complexity.
  \end{block}
\end{frame}

\begin{frame}{Elbow Method: Visual Example}
  \begin{center}
    \includegraphics[width=0.6\textwidth]{figures/clusteringgraph.pdf}
  \end{center}
  \begin{itemize}
    \item Cost vs.\ number of clusters.
    \item The elbow indicates a good choice of $k$ where additional clusters provide
          diminishing returns.
  \end{itemize}
\end{frame}

\begin{frame}{Limitations of the Elbow Method}
  \begin{alertblock}{Not always decisive}
    In some datasets the curve decreases \alert{smoothly} without a clear kink, making
    the ``elbow'' difficult to identify unambiguously.
  \end{alertblock}
  \vspace{0.4em}
  \begin{itemize}
    \item The elbow method is intuitive but subjective.
    \item No formal statistical criterion.
    \item Interpretation can vary across practitioners.
  \end{itemize}
  \vspace{0.4em}
  \begin{block}{Recommendation}
    Complement this heuristic with more principled, model-based criteria.
  \end{block}
\end{frame}

\begin{frame}{Information-Theoretic Criteria}
  \framesubtitle{Embedding $k$-means in a probabilistic framework}
  \begin{itemize}
    \item Although $k$-means is not inherently probabilistic, it can be embedded in a
          simple generative model.
    \item Assume each cluster generates points according to a spherical Gaussian distribution
          with fixed variance $\sigma^2$.
    \item This allows us to compute a \alert{likelihood} and apply information criteria.
  \end{itemize}
  \begin{block}{Akaike Information Criterion (AIC)}
    \[
    \text{AIC}(k) = 2p - 2\ln L,
    \]
    where $p$ is the number of parameters and $L$ is the likelihood.
  \end{block}
  \begin{block}{Bayesian Information Criterion (BIC)}
    \[
    \text{BIC}(k) = p\ln n - 2\ln L.
    \]
  \end{block}
\end{frame}

\begin{frame}{AIC and BIC for $k$-Means}
  \begin{itemize}
    \item For $k$-means embedded in a Gaussian model:
      \begin{itemize}
        \item Number of parameters: $p = k \cdot d + 1$ (prototypes + variance).
        \item Log-likelihood: proportional to $-J/(2\sigma^2)$ up to constants.
      \end{itemize}
    \item Both criteria penalize models with more parameters.
    \item \alert{BIC} imposes a stronger penalty when $n$ is large.
  \end{itemize}
  \begin{block}{Model selection rule}
    Choose $k$ that \alert{minimizes} AIC or BIC. Lower values indicate a preferable model
    according to the respective criterion.
  \end{block}
  \begin{alertblock}{Caveat}
    These criteria assume the probabilistic model is correct. For $k$-means, this is an
    approximation.
  \end{alertblock}
\end{frame}

\begin{frame}{Silhouette Coefficient}
  \framesubtitle{Measuring cluster quality}
  \begin{itemize}
    \item The \alert{silhouette coefficient} measures how well each point fits within its
          assigned cluster compared to other clusters.
    \item For point $\mathbf{x}_i$ in cluster $C_j$:
      \begin{itemize}
        \item $a(i)$ = average distance from $\mathbf{x}_i$ to other points in $C_j$.
        \item $b(i)$ = minimum average distance from $\mathbf{x}_i$ to points in any other
              cluster.
      \end{itemize}
    \item Silhouette coefficient for point $i$:
  \end{itemize}
  \[
  s(i) = \frac{b(i) - a(i)}{\max\{a(i), b(i)\}}.
  \]
  \begin{itemize}
    \item $s(i) \in [-1, 1]$: values near $+1$ indicate good clustering, near $-1$
          indicate misclassification.
  \end{itemize}
\end{frame}

\begin{frame}{Average Silhouette and Model Selection}
  \begin{itemize}
    \item The \alert{average silhouette coefficient} over all points:
  \end{itemize}
  \[
  \bar{s}(k) = \frac{1}{n}\sum_{i=1}^{n} s(i).
  \]
  \begin{block}{Model selection rule}
    Choose $k$ that \alert{maximizes} $\bar{s}(k)$. Higher values indicate better-defined,
    well-separated clusters.
  \end{block}
  \begin{itemize}
    \item Advantages:
      \begin{itemize}
        \item No probabilistic assumptions required.
        \item Interpretable: measures cluster cohesion and separation.
      \end{itemize}
    \item Disadvantages:
      \begin{itemize}
        \item Computationally expensive for large datasets ($O(n^2)$).
        \item Can be sensitive to outliers.
      \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}{Gap Statistic}
  \framesubtitle{Comparing to a null reference distribution}
  \begin{itemize}
    \item The \alert{gap statistic} compares the within-cluster dispersion of the data to
          that of a reference null distribution (e.g., uniform over feature space).
    \item Let $W_k$ be the within-cluster sum of squares for the actual data with $k$ clusters.
    \item Let $W_k^*$ be the expected value under the null distribution.
  \end{itemize}
  \[
  \text{Gap}(k) = \mathbb{E}[\log W_k^*] - \log W_k.
  \]
  \begin{block}{Model selection rule}
    Choose the \alert{smallest} $k$ such that $\text{Gap}(k) \geq \text{Gap}(k+1) - s_{k+1}$,
    where $s_{k+1}$ is the standard error of $\log W_{k+1}^*$.
  \end{block}
  \begin{itemize}
    \item Requires generating multiple reference datasets (bootstrap samples).
  \end{itemize}
\end{frame}

\begin{frame}{Practical Guidelines for Model Selection}
  \begin{itemize}
    \item No single method is universally best; combine multiple approaches:
      \begin{enumerate}
        \item Run elbow method for initial visual assessment.
        \item Compute AIC/BIC for probabilistic perspective.
        \item Calculate silhouette coefficient for cluster quality.
        \item Use gap statistic if computational budget allows.
      \end{enumerate}
    \item \alert{Domain knowledge} often provides strong constraints on plausible values of $k$.
    \item Visualize clusters (when $d$ is low or after dimensionality reduction) to validate
          that the chosen $k$ makes sense.
  \end{itemize}
  \begin{alertblock}{No substitute for interpretation}
    Automated criteria are helpful, but the final choice of $k$ should be informed by the
    specific goals of the analysis and expert judgment.
  \end{alertblock}
\end{frame}

\section{Hierarchical Clustering}

\begin{frame}{Introduction to Hierarchical Clustering}
  \framesubtitle{Building a tree of clusters}
  \begin{itemize}
    \item Hierarchical clustering does not produce a single partition.
    \item Instead, it builds a \alert{hierarchy} of partitions at different scales.
    \item Two main approaches:
      \begin{enumerate}
        \item \alert{Agglomerative} (bottom-up): Start with each point as its own cluster,
              repeatedly merge closest clusters.
        \item \alert{Divisive} (top-down): Start with all points in one cluster, repeatedly
              split clusters.
      \end{enumerate}
    \item Agglomerative methods are more common in practice.
  \end{itemize}
\end{frame}

\begin{frame}{Agglomerative Hierarchical Clustering}
  \framesubtitle{Bottom-up merging}
  \begin{enumerate}
    \item \textbf{Initialize}: Each of $n$ points is its own cluster.
    \item \textbf{Repeat} until one cluster remains:
      \begin{enumerate}
        \item Identify the two \alert{closest} clusters.
        \item Merge them into a single cluster.
        \item Record the merge in a dendrogram.
      \end{enumerate}
    \item \textbf{Output}: A dendrogram encoding the full merge history.
  \end{enumerate}
  \vspace{0.4em}
  \begin{block}{Key question}
    How do we measure the distance (or dissimilarity) between \alert{clusters} rather than
    individual points? This is determined by the \alert{linkage criterion}.
  \end{block}
\end{frame}

\begin{frame}{Linkage Criteria}
  \framesubtitle{Defining cluster distance}
  \begin{itemize}
    \item Let $C_i$ and $C_j$ be two clusters.
    \item Different linkage criteria define different notions of cluster distance:
  \end{itemize}
  \begin{block}{Single linkage (nearest neighbor)}
    \[
    d(C_i, C_j) = \min_{\mathbf{x}\in C_i, \mathbf{y}\in C_j} d(\mathbf{x}, \mathbf{y}).
    \]
    Distance between closest pair of points.
  \end{block}
  \begin{block}{Complete linkage (farthest neighbor)}
    \[
    d(C_i, C_j) = \max_{\mathbf{x}\in C_i, \mathbf{y}\in C_j} d(\mathbf{x}, \mathbf{y}).
    \]
    Distance between farthest pair of points.
  \end{block}
\end{frame}

\begin{frame}{Linkage Criteria (continued)}
  \begin{block}{Average linkage}
    \[
    d(C_i, C_j) = \frac{1}{|C_i||C_j|}\sum_{\mathbf{x}\in C_i}\sum_{\mathbf{y}\in C_j}
                  d(\mathbf{x}, \mathbf{y}).
    \]
    Average pairwise distance between points in the two clusters.
  \end{block}
  \begin{block}{Ward's linkage}
    Merge clusters that minimize the increase in within-cluster sum of squares (similar
    to $k$-means objective). Tends to produce balanced, compact clusters.
  \end{block}
  \begin{alertblock}{Impact of linkage choice}
    Different linkage criteria can produce very different dendrograms. Single linkage tends
    to produce elongated clusters (chaining), while complete linkage prefers compact clusters.
  \end{alertblock}
\end{frame}

\begin{frame}{The Dendrogram}
  \framesubtitle{Visualizing the hierarchy}
  \begin{itemize}
    \item A \alert{dendrogram} is a tree diagram showing the nested sequence of merges.
    \item The \alert{height} of each merge in the dendrogram represents the distance (or
          dissimilarity) at which the merge occurred.
    \item To obtain a flat partition with $k$ clusters: \alert{cut} the dendrogram at a
          horizontal line that intersects $k$ branches.
  \end{itemize}
  \begin{block}{Flexibility}
    A single dendrogram encodes all possible values of $k$ simultaneously. We can explore
    different granularities by cutting at different heights.
  \end{block}
\end{frame}

\begin{frame}{Example Dendrogram}
  \begin{center}
    \includegraphics[width=0.7\textwidth]{figures/iris2.pdf}
  \end{center}
  \begin{itemize}
    \item Dendrogram from hierarchical clustering on a dataset.
    \item Height of each merge indicates dissimilarity between merged clusters.
    \item Cutting at different heights yields different numbers of clusters.
  \end{itemize}
\end{frame}

\begin{frame}{Computational Complexity of Hierarchical Clustering}
  \begin{itemize}
    \item Naive agglomerative algorithm: $O(n^3)$ time complexity.
      \begin{itemize}
        \item $n-1$ merges.
        \item Each merge requires scanning all pairs of clusters to find the closest.
      \end{itemize}
    \item With efficient data structures (e.g., priority queues), can be reduced to $O(n^2\log n)$
          for certain linkage criteria.
    \item Space: $O(n^2)$ for storing the distance matrix.
  \end{itemize}
  \begin{alertblock}{Scalability limitation}
    Hierarchical clustering is generally more expensive than $k$-means, limiting its
    applicability to large datasets unless approximations or efficient variants are used.
  \end{alertblock}
\end{frame}

\begin{frame}{Properties of Hierarchical Clustering}
  \begin{block}{Advantages}
    \begin{itemize}
      \item No need to specify $k$ in advance.
      \item Dendrogram provides a rich multi-scale view of data structure.
      \item Deterministic: no random initialization, always produces the same result
            (given linkage and distance).
    \end{itemize}
  \end{block}
  \begin{alertblock}{Disadvantages}
    \begin{itemize}
      \item Higher computational cost.
      \item Greedy decisions: once a merge is made, it cannot be undone.
      \item Sensitive to choice of linkage criterion and distance metric.
      \item Can struggle with noisy data or outliers (especially single linkage).
    \end{itemize}
  \end{alertblock}
\end{frame}

\begin{frame}{Choosing the Number of Clusters from a Dendrogram}
  \begin{itemize}
    \item How to decide where to cut the dendrogram?
    \item Several heuristics:
      \begin{enumerate}
        \item \alert{Visual inspection}: Look for a large vertical gap in the dendrogram,
              suggesting a natural separation.
        \item \alert{Inconsistency criterion}: Cut where merge heights show large jumps.
        \item \alert{Elbow method on heights}: Plot merge heights and look for an elbow.
        \item \alert{Domain knowledge}: Use prior information about expected number of groups.
      \end{enumerate}
  \end{itemize}
  \begin{block}{Practical approach}
    Cutting the dendrogram at a height that yields genuinely dissimilar groups, as
    indicated by large merge heights, often produces meaningful clusters.
  \end{block}
\end{frame}

\section{Mixture Models}

\begin{frame}{From Hard to Soft Clustering}
  \framesubtitle{Introducing probabilistic cluster assignments}
  \begin{itemize}
    \item $k$-means produces \alert{hard assignments}: each point belongs to exactly one cluster.
    \item In many real-world scenarios, cluster boundaries are \alert{fuzzy} and points near
          boundaries could reasonably belong to multiple clusters.
    \item \alert{Mixture models} provide a principled probabilistic framework for
          \alert{soft clustering}.
  \end{itemize}
  \begin{block}{Key idea}
    Model the data as generated from a \alert{mixture} of probability distributions.
    Each distribution represents a cluster. Cluster membership is a latent random variable.
  \end{block}
\end{frame}

\begin{frame}{Gaussian Mixture Models (GMMs)}
  \framesubtitle{A generative model for clustering}
  \begin{itemize}
    \item A \alert{Gaussian Mixture Model} (GMM) assumes the data are generated by a
          mixture of $K$ Gaussian distributions.
    \item Generative process:
      \begin{enumerate}
        \item Choose a cluster $k$ with probability $\pi_k$ (mixing coefficient).
        \item Generate $\mathbf{x}$ from a Gaussian distribution $\mathcal{N}(\boldsymbol{\mu}_k,
              \boldsymbol{\Sigma}_k)$.
      \end{enumerate}
    \item Probability density of observing $\mathbf{x}$:
  \end{itemize}
  \[
  p(\mathbf{x}) = \sum_{k=1}^{K} \pi_k \mathcal{N}(\mathbf{x} \mid \boldsymbol{\mu}_k,
                  \boldsymbol{\Sigma}_k),
  \]
  where $\sum_{k=1}^{K}\pi_k = 1$ and $\pi_k \geq 0$.
\end{frame}

\begin{frame}{GMM Parameters}
  \begin{itemize}
    \item A GMM is characterized by:
      \begin{itemize}
        \item \alert{Mixing coefficients} $\pi_k$: probability of selecting component $k$.
        \item \alert{Means} $\boldsymbol{\mu}_k \in \mathbb{R}^d$: center of component $k$.
        \item \alert{Covariances} $\boldsymbol{\Sigma}_k$: shape and orientation of
              component $k$.
      \end{itemize}
    \item Parameters: $\boldsymbol{\theta} = \{\pi_k, \boldsymbol{\mu}_k,
          \boldsymbol{\Sigma}_k\}_{k=1}^{K}$.
    \item Goal: estimate $\boldsymbol{\theta}$ from data $\mathbf{X} = \{\mathbf{x}_1,
          \ldots, \mathbf{x}_n\}$.
  \end{itemize}
  \begin{block}{Maximum likelihood estimation}
    Find parameters that maximize the likelihood:
    \[
    L(\boldsymbol{\theta}) = \prod_{i=1}^{n} p(\mathbf{x}_i) = \prod_{i=1}^{n}
                             \sum_{k=1}^{K} \pi_k \mathcal{N}(\mathbf{x}_i \mid
                             \boldsymbol{\mu}_k, \boldsymbol{\Sigma}_k).
    \]
  \end{block}
\end{frame}

\begin{frame}{Latent Variables and Responsibilities}
  \begin{itemize}
    \item Introduce latent variable $z_i \in \{1, \ldots, K\}$ indicating which component
          generated $\mathbf{x}_i$.
    \item If $z_i$ were observed, MLE would be straightforward (like supervised learning).
    \item Since $z_i$ is latent, we compute the \alert{posterior probability} that
          $\mathbf{x}_i$ was generated by component $k$:
  \end{itemize}
  \[
  \gamma_k(\mathbf{x}_i) = p(z_i=k \mid \mathbf{x}_i, \boldsymbol{\theta})
                         = \frac{\pi_k \mathcal{N}(\mathbf{x}_i \mid \boldsymbol{\mu}_k,
                                 \boldsymbol{\Sigma}_k)}{\sum_{j=1}^{K} \pi_j
                                 \mathcal{N}(\mathbf{x}_i \mid \boldsymbol{\mu}_j,
                                 \boldsymbol{\Sigma}_j)}.
  \]
  \begin{block}{Responsibility}
    $\gamma_k(\mathbf{x}_i)$ is the \alert{responsibility} of component $k$ for point
    $\mathbf{x}_i$. It quantifies soft assignment: $\sum_{k=1}^{K}\gamma_k(\mathbf{x}_i)=1$.
  \end{block}
\end{frame}

\begin{frame}{The EM Algorithm for GMMs}
  \framesubtitle{Expectation-Maximization}
  \begin{itemize}
    \item Direct maximization of the log-likelihood is intractable due to the sum inside
          the logarithm.
    \item The \alert{Expectation-Maximization (EM)} algorithm provides an iterative solution.
    \item EM alternates between two steps:
  \end{itemize}
  \begin{block}{E-step (Expectation)}
    Compute responsibilities $\gamma_k(\mathbf{x}_i)$ for all $i$ and $k$ using current
    parameters $\boldsymbol{\theta}^{(t)}$.
  \end{block}
  \begin{block}{M-step (Maximization)}
    Update parameters $\boldsymbol{\theta}^{(t+1)}$ by maximizing the expected complete-data
    log-likelihood using the computed responsibilities.
  \end{block}
  \begin{itemize}
    \item Repeat until convergence.
  \end{itemize}
\end{frame}

\begin{frame}{EM Algorithm: E-Step}
  \begin{itemize}
    \item Given current parameters $\boldsymbol{\theta}^{(t)} = \{\pi_k^{(t)},
          \boldsymbol{\mu}_k^{(t)}, \boldsymbol{\Sigma}_k^{(t)}\}_{k=1}^{K}$:
    \item Compute responsibilities:
  \end{itemize}
  \[
  \gamma_k^{(t)}(\mathbf{x}_i) = \frac{\pi_k^{(t)} \mathcal{N}(\mathbf{x}_i \mid
                                       \boldsymbol{\mu}_k^{(t)},
                                       \boldsymbol{\Sigma}_k^{(t)})}{\sum_{j=1}^{K}
                                       \pi_j^{(t)} \mathcal{N}(\mathbf{x}_i \mid
                                       \boldsymbol{\mu}_j^{(t)},
                                       \boldsymbol{\Sigma}_j^{(t)})}.
  \]
  \begin{itemize}
    \item This is analogous to the assignment step in $k$-means, but now assignments are
          \alert{soft} (probabilistic) rather than hard.
  \end{itemize}
\end{frame}

\begin{frame}{EM Algorithm: M-Step}
  \begin{itemize}
    \item Update parameters using the computed responsibilities:
  \end{itemize}
  \begin{block}{Effective number of points in cluster $k$}
    \[
    n_k = \sum_{i=1}^{n} \gamma_k^{(t)}(\mathbf{x}_i).
    \]
  \end{block}
  \begin{block}{Update equations}
    \[
    \pi_k^{(t+1)} = \frac{n_k}{n}, \quad
    \boldsymbol{\mu}_k^{(t+1)} = \frac{1}{n_k}\sum_{i=1}^{n} \gamma_k^{(t)}(\mathbf{x}_i)
                                 \mathbf{x}_i,
    \]
    \[
    \boldsymbol{\Sigma}_k^{(t+1)} = \frac{1}{n_k}\sum_{i=1}^{n} \gamma_k^{(t)}(\mathbf{x}_i)
                                    (\mathbf{x}_i - \boldsymbol{\mu}_k^{(t+1)})
                                    (\mathbf{x}_i - \boldsymbol{\mu}_k^{(t+1)})^T.
    \]
  \end{block}
  \begin{itemize}
    \item These are \alert{responsibility-weighted} sample statistics.
  \end{itemize}
\end{frame}

\begin{frame}{EM Convergence Properties}
  \begin{itemize}
    \item The EM algorithm is guaranteed to \alert{monotonically increase} the log-likelihood
          at each iteration (or leave it unchanged).
    \item Convergence is to a \alert{local maximum}, not necessarily global.
    \item Different initializations can lead to different solutions.
  \end{itemize}
  \begin{block}{Initialization strategies}
    \begin{itemize}
      \item Run $k$-means first and use resulting clusters to initialize GMM parameters.
      \item Multiple random starts and select the solution with highest likelihood.
      \item Use hierarchical clustering or domain knowledge to initialize.
    \end{itemize}
  \end{block}
  \begin{alertblock}{Convergence criterion}
    Stop when the change in log-likelihood is below a threshold, or when parameters
    change negligibly.
  \end{alertblock}
\end{frame}

\begin{frame}{Connection Between $k$-Means and GMMs}
  \begin{itemize}
    \item $k$-means can be viewed as a \alert{special case} of GMM:
      \begin{itemize}
        \item All covariances $\boldsymbol{\Sigma}_k = \sigma^2 \mathbf{I}$ (spherical,
              equal variance).
        \item All mixing coefficients $\pi_k = 1/K$ (equal prior).
        \item Take the limit $\sigma^2 \to 0$: soft assignments become hard.
      \end{itemize}
    \item In this limit, the EM algorithm for GMM reduces to Lloyd's algorithm for $k$-means.
  \end{itemize}
  \begin{block}{Insight}
    $k$-means is a degenerate GMM with fixed, isotropic covariances and hard assignments.
    GMMs generalize this by allowing flexible covariance structures and soft assignments.
  \end{block}
\end{frame}

\begin{frame}{Poisson Mixture Models}
  \framesubtitle{Extending to count data}
  \begin{itemize}
    \item The mixture framework is not limited to continuous data.
    \item For \alert{count} observations, a Poisson distribution is natural.
    \item Univariate Poisson component:
  \end{itemize}
  \[
  q(x\mid\lambda_k)=\frac{e^{-\lambda_k}\lambda_k^x}{x!},
  \]
  where $\lambda_k>0$ is the rate parameter of component $k$.
  \begin{itemize}
    \item The EM algorithm for Poisson mixtures mirrors the Gaussian case:
      \begin{itemize}
        \item E-step: Compute responsibilities.
        \item M-step: Update $\pi_k$ and $\lambda_k$ using responsibility-weighted counts.
      \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}{Poisson Mixture: EM Updates}
  \begin{itemize}
    \item E-step (identical in structure to Gaussian case):
  \end{itemize}
  \[
  \gamma_k^{(t)}(x_i)=
  \frac{\pi_k^{(t)}\,q(x_i\mid\lambda_k^{(t)})}{\sum_{j=1}^{K}\pi_j^{(t)}
        \,q(x_i\mid\lambda_j^{(t)})}.
  \]
  \begin{itemize}
    \item M-step: Mixing coefficients update as before, $\pi_k^{(t+1)}=n_k/n$.
    \item Rate parameter update:
  \end{itemize}
  \[
  \lambda_k^{(t+1)}=
  \frac{\sum_{i=1}^{n}\gamma_k^{(t)}(x_i)\,x_i}{\sum_{i=1}^{n}\gamma_k^{(t)}(x_i)}
  =\frac{1}{n_k}\sum_{i=1}^{n}\gamma_k^{(t)}(x_i)\,x_i.
  \]
  \begin{itemize}
    \item The updated rate is the \alert{responsibility-weighted} sample mean of the counts.
  \end{itemize}
\end{frame}

\begin{frame}{Model Selection for Mixture Models}
  \framesubtitle{Choosing the number of components $K$}
  \begin{itemize}
    \item Once mixture models have been fitted for different values of $K$, we need to
          decide how many components provide an adequate description.
    \item Because mixture models define a likelihood, we can use \alert{information criteria}.
  \end{itemize}
  \begin{block}{Akaike Information Criterion (AIC)}
    \[
    \text{AIC}(K)=2\cdot(\text{number of parameters})-2\ln L.
    \]
  \end{block}
  \begin{block}{Bayesian Information Criterion (BIC)}
    \[
    \text{BIC}(K)=(\text{number of parameters})\cdot\ln n-2\ln L.
    \]
  \end{block}
  \begin{itemize}
    \item Both penalize models with more parameters; BIC imposes a stronger penalty when
          $n$ is large.
  \end{itemize}
\end{frame}

\begin{frame}{Number of Parameters in a GMM}
  \begin{itemize}
    \item For a GMM with $K$ components in $d$ dimensions:
      \begin{itemize}
        \item $K$ mean vectors: $K \cdot d$ parameters.
        \item $K$ covariance matrices: $K \cdot d(d+1)/2$ parameters (if full covariance).
        \item $K$ mixing coefficients: $K-1$ free parameters (one constraint: $\sum\pi_k=1$).
      \end{itemize}
    \item Total: $K\big(d+d(d+1)/2\big)+K-1$ parameters.
  \end{itemize}
  \begin{block}{Model selection rule}
    Lower values of AIC or BIC indicate a preferable model. Fit GMMs for a range of $K$,
    compute AIC/BIC for each, and select the $K$ that minimizes the criterion.
  \end{block}
  \begin{alertblock}{Asymptotic consistency of BIC}
    Under suitable regularity conditions and correct model specification, BIC selects the
    true number of components as $n\to\infty$.
  \end{alertblock}
\end{frame}

\begin{frame}{Advantages of Information Criteria for Mixtures}
  \begin{itemize}
    \item Key advantage over heuristics used for $k$-means:
      \begin{itemize}
        \item Information criteria are \alert{derived} from a coherent statistical model
              rather than from geometric intuition.
        \item Minimizing AIC or BIC provides a principled answer to the question of how
              many components are needed.
      \end{itemize}
    \item BIC in particular has strong theoretical properties.
  \end{itemize}
  \begin{block}{Practical workflow}
    \begin{enumerate}
      \item Fit GMMs for $K=1,2,\ldots,K_{\max}$.
      \item Compute AIC and BIC for each $K$.
      \item Plot AIC/BIC vs.\ $K$ and select the minimizer.
      \item Validate with domain knowledge and qualitative inspection.
    \end{enumerate}
  \end{block}
\end{frame}

\section{GMM vs.\ $k$-Means: Key Differences}

\begin{frame}{GMM vs.\ $k$-Means: Conceptual Differences}
  \begin{itemize}
    \item Although both are widely used for discovering structure in unlabeled data, they
          embody two rather different views of what a cluster is and how it should be inferred.
  \end{itemize}
  \begin{columns}[t]
    \column{0.48\textwidth}
      \begin{block}{$k$-Means}
        \begin{itemize}
          \item Cluster as a \alert{geometric region}.
          \item \alert{Hard partition} of the dataset.
          \item No representation of ambiguity.
          \item Every point belongs to exactly one cluster.
        \end{itemize}
      \end{block}
    \column{0.48\textwidth}
      \begin{block}{GMM}
        \begin{itemize}
          \item Cluster as a \alert{probabilistic component}.
          \item \alert{Soft assignments} via responsibilities.
          \item Explicit quantification of uncertainty.
          \item Points can have fractional membership.
        \end{itemize}
      \end{block}
  \end{columns}
\end{frame}

\begin{frame}{GMM vs.\ $k$-Means: Learning Objectives}
  \begin{columns}[t]
    \column{0.48\textwidth}
      \begin{block}{$k$-Means}
        \begin{itemize}
          \item Minimizes a purely \alert{geometric criterion}.
          \item Within-cluster sum of squares $J$.
          \item Encourages compact clusters around centroids.
          \item Distance-based cost.
        \end{itemize}
      \end{block}
    \column{0.48\textwidth}
      \begin{block}{GMM}
        \begin{itemize}
          \item Maximizes the \alert{likelihood} $L$ under a generative model.
          \item Likelihood-based criterion.
          \item Fits a probabilistic distribution to the data.
          \item Enables principled model comparison (AIC, BIC).
        \end{itemize}
      \end{block}
  \end{columns}
  \vspace{0.6em}
  \begin{alertblock}{Consequence for model selection}
    $k$-means relies on heuristics (elbow, silhouette); GMMs can be compared using
    information criteria that explicitly balance fit against complexity.
  \end{alertblock}
\end{frame}

\begin{frame}{GMM vs.\ $k$-Means: Cluster Shapes}
  \begin{itemize}
    \item Another important distinction: shapes of clusters each method can represent.
  \end{itemize}
  \begin{columns}[t]
    \column{0.48\textwidth}
      \begin{block}{$k$-Means}
        \begin{itemize}
          \item Tends to produce \alert{spherical} clusters of comparable scale.
          \item Relies on Euclidean distances to centroids.
          \item Cannot model elongated or oriented clusters well.
        \end{itemize}
      \end{block}
    \column{0.48\textwidth}
      \begin{block}{GMM}
        \begin{itemize}
          \item Each component has its own \alert{covariance matrix} $\boldsymbol{\Sigma}_k$.
          \item Clusters can be \alert{elliptical, oriented}, and of different sizes/densities.
          \item Much more flexible representation.
        \end{itemize}
      \end{block}
  \end{columns}
  \vspace{0.6em}
  \begin{alertblock}{Computational cost}
    This added flexibility comes at a price: GMM iterations are more expensive, especially
    with full covariances in high dimensions.
  \end{alertblock}
\end{frame}

\begin{frame}{GMM vs.\ $k$-Means: Computational Efficiency}
  \begin{columns}[t]
    \column{0.48\textwidth}
      \begin{block}{$k$-Means}
        \begin{itemize}
          \item Updates are \alert{fast}.
          \item Typically requires only a few iterations.
          \item Scales well to large datasets.
          \item Often used as initialization for GMMs.
        \end{itemize}
      \end{block}
    \column{0.48\textwidth}
      \begin{block}{GMM}
        \begin{itemize}
          \item EM iterations are \alert{more expensive}.
          \item Involves computing Gaussian densities and inverting covariance matrices.
          \item May converge more slowly.
          \item Higher memory requirements.
        \end{itemize}
      \end{block}
  \end{columns}
\end{frame}

\begin{frame}{GMM vs.\ $k$-Means: Interpretability}
  \begin{itemize}
    \item The two methods differ in interpretability.
  \end{itemize}
  \begin{block}{$k$-Means}
    Best understood as a \alert{geometric partition} induced by distances to prototypes.
    Simple and intuitive: each cluster is a region of space around a centroid.
  \end{block}
  \begin{block}{GMM}
    Provides a fully \alert{generative probabilistic description} of the data. Clusters
    are not just regions of space but components of a mixture distribution. Assignments are
    naturally probabilistic, with responsibilities quantifying uncertainty.
  \end{block}
  \vspace{0.4em}
  \begin{itemize}
    \item GMMs particularly appealing when uncertainty matters, when cluster shapes are
          not spherical, or when one wants likelihood-based tools for comparing models with
          different values of $K$.
  \end{itemize}
\end{frame}

\section{Practical Considerations}

\begin{frame}{When to Use Mixture Models}
  \begin{block}{Mixture models are especially appealing when:}
    \begin{itemize}
      \item You need more than a partition: \alert{uncertainty-aware membership}.
      \item You want a coherent \alert{probabilistic interpretation}.
      \item You need \alert{likelihood-based tools} for model comparison.
      \item Clusters have different volumes, orientations, or shapes.
      \item You want to adapt to \alert{non-Gaussian data} by choosing appropriate
            component families (e.g., Poisson, multinomial).
    \end{itemize}
  \end{block}
\end{frame}

\begin{frame}{Costs and Challenges of Mixture Models}
  \begin{alertblock}{Computational demands}
    EM is more computationally demanding than $k$-means. Convergence to a local optimum
    makes initialization important; multiple random starts are often necessary.
  \end{alertblock}
  \vspace{0.4em}
  \begin{alertblock}{Degeneracies}
    In GMMs with full covariances, must be aware of near-singular covariance matrices,
    which can arise when a component collapses around very few points.
  \end{alertblock}
  \vspace{0.4em}
  \begin{alertblock}{Data requirements}
    Richer models carry more parameters, so reliable estimation requires \alert{sufficient
    data}. The chosen component family must be a reasonable approximation to the true
    data-generating process.
  \end{alertblock}
\end{frame}

\begin{frame}{Practical Workflow}
  \begin{block}{Common workflow reflecting trade-offs:}
    \begin{enumerate}
      \item Begin with \alert{$k$-means} as a fast exploratory tool and as a source of
            good initializations for EM.
      \item Move to \alert{mixture models} when soft assignments or a likelihood-based
            interpretation are needed.
      \item \alert{Standardize features} before fitting Gaussian components (especially
            when features have different scales).
      \item Compare several values of $K$ through \alert{AIC and BIC}, complementing
            these criteria with domain knowledge and qualitative inspection of the resulting
            clusters.
      \item Visualize clusters (when feasible) to validate that the chosen model makes sense.
    \end{enumerate}
  \end{block}
\end{frame}

\begin{frame}{Feature Standardization}
  \begin{itemize}
    \item When features have different scales, distances are dominated by high-variance
          features.
    \item This can severely distort clustering results.
  \end{itemize}
  \begin{block}{Standardization}
    Transform each feature to have zero mean and unit variance:
    \[
    \tilde{x}_{ij} = \frac{x_{ij} - \bar{x}_j}{\sigma_j},
    \]
    where $\bar{x}_j$ and $\sigma_j$ are the mean and standard deviation of feature $j$.
  \end{block}
  \begin{alertblock}{When to standardize}
    \begin{itemize}
      \item Almost always for $k$-means and GMM with Euclidean distance.
      \item Especially important when features have different units or scales.
    \end{itemize}
  \end{alertblock}
\end{frame}

\begin{frame}{Handling High-Dimensional Data}
  \begin{itemize}
    \item Clustering in high dimensions faces the \alert{curse of dimensionality}.
    \item Distances become less meaningful as $d$ grows.
    \item Computational cost increases.
  \end{itemize}
  \begin{block}{Strategies}
    \begin{itemize}
      \item \alert{Dimensionality reduction}: Apply PCA, t-SNE, or UMAP before clustering.
      \item \alert{Feature selection}: Identify and use only the most informative features.
      \item \alert{Regularization}: Use diagonal or spherical covariances in GMMs to reduce
            parameters.
      \item \alert{Subspace clustering}: Methods that cluster in different subspaces of
            the feature space.
    \end{itemize}
  \end{block}
\end{frame}

\begin{frame}{Dealing with Outliers}
  \begin{itemize}
    \item Outliers can severely affect clustering results.
    \item $k$-means is particularly sensitive: a single outlier can drag a centroid away
          from the true cluster center.
    \item Hierarchical clustering with single linkage can suffer from chaining.
  \end{itemize}
  \begin{block}{Strategies}
    \begin{itemize}
      \item \alert{Robust variants}: Use $k$-medoids instead of $k$-means (medoids are
            less sensitive to outliers).
      \item \alert{Outlier detection}: Identify and remove outliers before clustering.
      \item \alert{Mixture models with noise component}: Add a uniform or diffuse component
            to capture outliers.
      \item \alert{Trimming}: Remove a small fraction of most distant points before fitting.
    \end{itemize}
  \end{block}
\end{frame}

  \begin{center}
  \begin{tabular}{ll}

    \textbf{Concept} & \textbf{Definition / Key Idea} \\
    \midrule

    Lloyd's algorithm & Alternating assignment and update steps \\
    Convergence & Monotonic decrease of $J$, local optimum \\

    Model selection & Elbow, silhouette, AIC, BIC, gap statistic \\
    Hierarchical & Dendrogram, linkage criteria, $O(n^3)$ \\
    GMM & $p(\mathbf{x}) = \sum_k \pi_k \mathcal{N}(\mathbf{x} \mid \boldsymbol{\mu}_k, \boldsymbol{\Sigma}_k)$ \\
    EM algorithm & E-step (responsibilities) + M-step (parameters) \\
    Soft assignment & $\gamma_k(\mathbf{x}_i) = p(z_i=k \mid \mathbf{x}_i)$ \\
    \bottomrule
  \end{tabular}
  \end{center}
\end{frame}

\begin{frame}{Summary: Methods at a Glance}
  \begin{columns}[t]
    \column{0.48\textwidth}
      \begin{block}{$k$-Means}
        \begin{itemize}
          \item Fast, scalable partitional method.
          \item Hard assignments, spherical clusters.
          \item Lloyd's algorithm: $O(nkdt)$.
          \item Initialization matters ($k$-means++).
          \item Heuristic model selection (elbow, silhouette).
        \end{itemize}
      \end{block}
      \vspace{0.4em}
      \begin{block}{Hierarchical}
        \begin{itemize}
          \item Dendrogram: nested partitions.
          \item No need to specify $k$ upfront.
          \item Linkage criteria: single, complete, average, Ward.
          \item $O(n^3)$ complexity, greedy decisions.
        \end{itemize}
      \end{block}
    \column{0.48\textwidth}
      \begin{block}{Gaussian Mixtures}
        \begin{itemize}
          \item Probabilistic, soft assignments.
          \item Flexible cluster shapes (elliptical).
          \item EM algorithm for parameter estimation.
          \item Likelihood-based model selection (AIC, BIC).
          \item More expensive than $k$-means.
        \end{itemize}
      \end{block}
      \vspace{0.4em}
      \begin{block}{Practical notes}
        \begin{itemize}
          \item Standardize features.
          \item Use $k$-means for initialization.
          \item Combine multiple criteria for model selection.
          \item Validate with domain knowledge.
        \end{itemize}
      \end{block}
  \end{columns}
\end{frame}

\begin{frame}{Summary: Choosing a Method}
  \renewcommand{\arraystretch}{1.4}
  \begin{center}
  \begin{tabular}{lll}
    \toprule
    \textbf{Scenario} & \textbf{Recommended approach} \\
    \midrule
    Large dataset, speed matters & $k$-means \\
    Uncertainty quantification needed & GMM with EM \\
    Unknown $k$, explore multi-scale & Hierarchical clustering \\
    Non-spherical clusters & GMM with full covariance \\
    Count data & Poisson mixture model \\
    High dimensions & PCA + $k$-means, or subspace methods \\
    Outliers present & $k$-medoids, or robust variants \\
    \bottomrule
  \end{tabular}
  \end{center}
\end{frame}

\begin{frame}{Final Remarks}
  \begin{itemize}
    \item Clustering is a fundamental tool in unsupervised learning, with applications
          across all domains of data science.
    \item No single clustering method is universally best; the appropriate choice depends
          on:
      \begin{itemize}
        \item The structure of the data.
        \item The goals of the analysis.
        \item Computational constraints.
        \item The need for interpretability vs.\ flexibility.
      \end{itemize}
    \item Combining multiple methods and criteria, and validating with domain knowledge,
          leads to the most robust and insightful results.
  \end{itemize}
  \begin{block}{Key takeaway}
    Understanding the inductive biases, strengths, and limitations of different clustering
    approaches is essential for selecting and applying them effectively.
  \end{block}
\end{frame}
